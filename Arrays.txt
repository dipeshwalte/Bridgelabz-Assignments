3. Extend the Prime Factorization Program to store all the Prime Factors of a
number n into an array and finally display the output.
	#!/bin/bash
echo Enter a number
read num
isPrime(){
   if [ $1 == 2 ]
   then
      flag=true
      return
   fi
   for((j=2;j<$1;j++))
   do
      if [ $(($1%j)) -eq 0 ]
      then
      flag=false
      break
      fi
   done
}
counter=0
for((i=2;i<=$num;i++))
do
   if [ $(($num%i)) -eq 0 ]
   then
      flag=true
      isPrime $i
      if [ $flag == true ]
      then
         array[$((counter++))]=$i
      fi
      flag=false
   fi
done
echo Prime factors are - ${array[@]}




4. Write a Program to show Sum of three Integer adds to ZERO
	#!/bin/bash
for((i=0;i<3;i++))
do
   echo Enter a number
   read num
   array[i]=$num
done
sum=0
for((i=0;i<3;i++))
do
   sum=$((${array[i]}+$sum))
done
if [ $sum -eq 0 ]
then
   echo sum of 3 numbers adds upto 0
else
   echo sum of 3 numbers is not 0, it is $sum
